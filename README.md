## Install

```bash
npm install --save word-cloud-d3-react
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'word-cloud-d3-react'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
